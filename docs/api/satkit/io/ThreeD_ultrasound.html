 56def generate_recording_list(directory):
 57    """
 58    Produce an array of Recordings from a 3D4D ultrasound directory.
 59
 60    Prepare a list of Recording objects from the files exported by AAA
 61    into the named directory. File existence is tested for,
 62    and if crucial files are missing from a given recording it will be
 63    excluded.
 64
 65    If problems are found with a recording, exclusion is marked with
 66    recordingObjet.excluded rather than not listing the recording. Log
 67    file will show reasons of exclusion.
 68
 69    The processed files are
 70    ultrasound and corresponding meta: .DCM, and
 71    audio waveform: .dat or .wav.
 72
 73    Additionally this will be added, but missing files are considered
 74    non-fatal:
 75    TextGrid: .textgrid.
 76
 77    Positional argument:
 78    directory -- the path to the directory to be processed.
 79    Returns an array of Recording objects sorted by date and time
 80        of recording.
 81    """
 82
 83    dicom_dir = directory / "DICOM"
 84    note_dir = directory / "NOTES"
 85    directories = {
 86        'root_dir': directory,
 87        'dicom_dir': dicom_dir,
 88        'note_dir': note_dir,
 89    }
 90
 91    dicom_files = sorted(dicom_dir.glob('*.DCM'))
 92    mat_file = list(note_dir.glob('officialNotes*.mat'))[0]
 93
 94    # strip file extensions off of filepaths to get the base names
 95    dicom_basenames = [filepath.name for filepath in dicom_files]
 96    meta = ThreeD_UltrasoundRecording.readMetaFromMat(mat_file)
 97
 98    # Create a lookup table for matching sound and dicom.
 99    # First file names with their ordinal numbers.
100    dicom_names_numbers = [[name, name.split('_')[1]]
101                           for name in dicom_basenames]
102    # Then a dict mapping number strings to filenames.
103    dicom_dict = {
104        name_number[1].split('.')[0]: name_number[0]
105        for name_number in dicom_names_numbers
106    }
107
108    recordings = []
109    for token in meta:
110        if token['trial_number'] in dicom_dict:
111            print(dicom_dict[token['trial_number']])
112            recording = generateUltrasoundRecording(
113                dicom_dict[token['trial_number']],
114                token['dat_filename'],
115                directories)
116            recording.addMeta(token)
117            recordings.append(recording)
118        else:
119            _3D4D_ultra_logger.info(
120                'No DICOM file corresponding to number ' +
121                token['trial_number'] + ' found in ' + str(directory) + '.')
122
123    return sorted(recordings, key=lambda token: token.meta['date_and_time'])

231def generateUltrasoundRecording(dicom_name, sound_name, directories=None):
232    """
233    Generate an UltrasoundRecording without Modalities.
234
235    Arguments:
236    dicom_name -- name of the DICOM files to be read without type 
237        extensions but with path.
238    sound_name -- name of the sound files (.dat and .wav) to be read 
239        without type extensions but with path.
240
241    KeywordArguments:
242    directory -- path to files
243
244    Returns an ThreeD_UltrasoundRecording without any modalities.
245    """
246
247    _3D4D_ultra_logger.info(
248        "Building Recording object for " + str(dicom_name) + " in " +
249        str(directories['root_dir']) + ".")
250
251    recording = ThreeD_UltrasoundRecording(
252        paths=directories,
253        basename=dicom_name,
254        sound_name=sound_name
255    )
256
257    # If we aren't going to process this recording,
258    # don't bother with the rest.
259    if recording.excluded:
260        _3D4D_ultra_logger.info(
261            "Recording " + str(dicom_name) + " automatically excluded.")
262
263    return recording

292    def _getData(self):
293        ds = pydicom.dcmread(self.meta['filename'])
294
295        # There are other options, but we don't deal with them just yet.
296        # Before 1.0: fix the above. see loadPhillipsDCM.m on how.
297        if len(ds.SequenceOfUltrasoundRegions) == 3:
298            type = ds[0x200d, 0x3016][1][0x200d, 0x300d].value
299            if type == 'UDM_USD_DATATYPE_DIN_3D_ECHO':
300                self._read_3D_ultra(ds)
301            else:
302                _3D4D_ultra_logger.critical(
303                    "Unknown DICOM ultrasound type: " + type + " in "
304                    + self.meta['filename'] + ".")
305                _3D4D_ultra_logger.critical('Exiting.')
306                sys.exit()
307        else:
308            _3D4D_ultra_logger.critical(
309                "Do not know what to do with data with "
310                + str(len(ds.SequenceOfUltrasoundRegions)) + " regions in "
311                + self.meta['filename'] + ".")
312            _3D4D_ultra_logger.critical('Exiting.')
313            sys.exit()
314
315        # Before 1.0: 'NumVectors' and 'PixPerVector' are bad names here.
316        # They come from the AAA ultrasound side of things and should be
317        # replaced, but haven't been yet as I'm in a hurry to get PD
318        # running on 3d4d ultrasound.
319        self.meta['no_frames'] = self.data.shape[0]
320        self.meta['NumVectors'] = self.data.shape[1]
321        self.meta['PixPerVector'] = self.data.shape[2]
322        ultra3D_time = np.linspace(
323            0, self.meta['no_frames'],
324            num=self.meta['no_frames'],
325            endpoint=False)
326        self.timevector = ultra3D_time / \
327            self.meta['FramesPerSec'] + self.timeOffset
328
329    def _read_3D_ultra(self, ds):
330        ultra_sequence = ds[0x200d, 0x3016][1][0x200d, 0x3020][0]
331
332        # data dimensions
333        numberOfFrames = int(ultra_sequence[0x200d, 0x3010].value)
334        shape = [int(token) for token in ds[0x200d, 0x3301].value]
335        frameSize = np.prod(shape)
336        shape.append(numberOfFrames)
337
338        # data scale in real space-time
339        scale = [float(token) for token in ds[0x200d, 0x3303].value]
340        self.meta['FramesPerSec'] = float(ds[0x200d, 0x3207].value)
341
342        # Before 1.0: unify the way scaling works across the data.
343        # here we have an attribute, in AAA ultrasound we have meta
344        # keys.
345        self._scale = scale
346
347        # The starting index for the non-junk data
348        # no junk from pydicom at the beginning. only at end of each frame
349        # s = 32
350
351        # Get the number of junk data points between each frame
352        interval = (
353            len(ultra_sequence[0x200d, 0x300e].value)-frameSize*numberOfFrames)
354        interval = int(interval/numberOfFrames)
355
356        data = np.frombuffer(
357            ultra_sequence[0x200d, 0x300e].value, dtype=np.uint8)
358
359        data.shape = (numberOfFrames, frameSize+interval)
360        data = np.take(data, np.arange(frameSize)+32, axis=1)
361        shape.reverse()
362        data.shape = shape
363
364        self._data = data
365
366        # this should be unnecessary now
367        # self._data = np.transpose(data)
368

402    @staticmethod
403    def readMetaFromMat(mat_file):
404        """
405        Read a WASL .mat file and return relevant contents as a dict.
406
407        Positional argument:
408        mat_file -- either a pathlib Path object representing the .mat 
409            file or a string of the same.
410
411        Returns -- an array of dicts that contain the following fields:
412            'trial_number': number of the recording within this session,
413            'prompt': prompt displayed to the participant,
414            'date_and_time': a datetime object of the time recording 
415                started, and
416            'dat_filename': string representing the name of the .dat 
417                sound file.
418        """
419        mat = scipy.io.loadmat(str(mat_file), squeeze_me=True)
420        meta = []
421        for element in mat['officialNotes']:
422            # Apparently squeeze_me=True is a bit too strident and
423            # somehow looses the shape of the most interesting level
424            # in the loadmat call. Not using it is not a good idea
425            # though so we do this:
426            element = element.item()
427            if len(element) > 5:
428                # We try this two ways, because at least once filename
429                # and date fields were in reversed order inside the
430                # .mat file.
431                try:
432                    date_and_time = datetime.strptime(
433                        element[4], "%d-%b-%Y %H:%M:%S")
434                    file_path = element[5]
435                except ValueError:
436                    date_and_time = datetime.strptime(
437                        element[5], "%d-%b-%Y %H:%M:%S")
438                    file_path = element[4]
439
440                meta_token = {
441                    'trial_number': element[0],
442                    'prompt': element[1],
443                    'date_and_time': date_and_time,
444                    'dat_filename': PureWindowsPath(file_path).name
445                }
446                meta.append(meta_token)
447        return meta

449    @staticmethod
450    def generateMeta(rows):
451        """
452        Read a WASL .mat file and return relevant contents as a dict.
453
454        Positional argument:
455        mat_file -- either a pathlib Path object representing the .mat
456            file or a string of the same.
457
458        Returns -- an array of dicts that contain the following fields:
459            'trial_number': number of the recording within this session,
460            'prompt': prompt displayed to the participant,
461            'date_and_time': a datetime object of the time recording
462                started, and
463            'dat_filename': string representing the name of the .dat
464                sound file.
465        """
466        meta = []
467        for row in rows:
468            date_and_time = datetime.strptime(
469                row['US File Name'], "%Y%m%d%H%M%S")
470
471            meta_token = {
472                'trial_number': row['number_portion'],
473                'prompt': row['Stimulus'],
474                'date_and_time': date_and_time,
475                'dat_filename': row['sound_filename'],
476                'sound_filename': row['sound_filename']
477            }
478            meta.append(meta_token)
479        return meta

551    def addMeta(self, meta):
552        """
553        Update self.meta with only the required key-value pairs.
554
555        The keys in meta are checked against
556        ThreeD_UltrasoundRecording.requiredMetaKeys.
557        It is a fatal error to not provide a value for all of those keys.
558        Any extra key-value pairs are discarded.
559
560        Positional argument:
561        meta -- a dict containing metadata.
562
563        Returns None.
564        """
565        if meta != None:
566            try:
567                wanted_meta = {
568                    key: meta[key]
569                    for key in ThreeD_UltrasoundRecording.requiredMetaKeys}
570            except KeyError:
571                # Missing metadata for one recording may be ok and this
572                # could be handled with just a call to
573                # _recording_logger.critical and setting
574                # self.excluded = True.
575                notFound = set(
576                    ThreeD_UltrasoundRecording.requiredMetaKeys) - set(meta)
577                _3D4D_ultra_logger.critical(
578                    "Part of metadata missing when processing " + self.meta
579                    ['filename'] + ". ")
580                _3D4D_ultra_logger.critical(
581                    "Could not find " + str(notFound) + ".")
582                _3D4D_ultra_logger.critical('Exiting.')
583                sys.exit()
584
585            self.meta.update(wanted_meta)

587    def addModalities(self, wavPreload=True, ultPreload=False,

591

595        ultPreload -- boolean indicating if the .ult file is to be read into
596            memory on initialising. Defaults to False. Note: these
597            files are, roughly one to two orders of magnitude
598            larger than .wav files.
599        videoPreload -- boolean indicating if the .avi file is to be read into
600            memory on initialising. Defaults to False. Note: these
601            files are, yet again, roughly one to two orders of magnitude
602            larger than .ult files.
603
604        Throws KeyError if TimeInSecsOfFirstFrame is missing from the 
605        meta file: [directory]/basename + .txt.
606        """
607        _3D4D_ultra_logger.info("Adding modalities to recording for "
608                                + self.meta['basename'] + ".")
609
610        # TODO: need to make sure that beep detect is not run.
611        waveform = MonoAudio(
612            parent=self,
613            preload=wavPreload,
614            time_offset=0,
615            filename=self.meta['ult_wav_file']
616        )
617        self.addModality(MonoAudio.__name__, waveform)
618        _3D4D_ultra_logger.debug(
619            "Added MonoAudio to Recording representing "
620            + self.meta['basename'] + ".")
621
622        # ultMeta = parseUltrasoundMetaAAA(recording.meta['ult_meta_file'])
623
624        ultrasound = ThreeD_Ultrasound(
625            parent=self,
626            preload=ultPreload,
627            filename=self.meta['ult_file']
628        )
629        self.addModality(ThreeD_Ultrasound.__name__, ultrasound)
630        _3D4D_ultra_logger.debug(
631            "Added RawUltrasound to Recording representing "
632            + self.meta['basename'] + ".")
633
634        if self.meta['video_exists']:
635            # This is the correct value for fps for a de-interlaced
636            # video for AAA recordings. Check it for other data.
637            videoMeta = {
638                'FramesPerSec': 59.94

641                "Video (.avi) fps set to " + str(videoMeta['FramesPerSec']) +

643                + " for AAA recordings. Check it for other data.")
644
645            video = Video(
646                parent=self,
647                preload=videoPreload,
648                path=self.meta['video_file'],

404    def readMetaFromMat(mat_file):
405        """
406        Read a WASL .mat file and return relevant contents as a dict.
407
408        Positional argument:
409        mat_file -- either a pathlib Path object representing the .mat 
410            file or a string of the same.
411
412        Returns -- an array of dicts that contain the following fields:
413            'trial_number': number of the recording within this session,
414            'prompt': prompt displayed to the participant,
415            'date_and_time': a datetime object of the time recording 
416                started, and
417            'dat_filename': string representing the name of the .dat 
418                sound file.
419        """
420        mat = scipy.io.loadmat(str(mat_file), squeeze_me=True)
421        meta = []
422        for element in mat['officialNotes']:
423            # Apparently squeeze_me=True is a bit too strident and
424            # somehow looses the shape of the most interesting level
425            # in the loadmat call. Not using it is not a good idea
426            # though so we do this:
427            element = element.item()
428            if len(element) > 5:
429                # We try this two ways, because at least once filename
430                # and date fields were in reversed order inside the
431                # .mat file.
432                try:
433                    date_and_time = datetime.strptime(
434                        element[4], "%d-%b-%Y %H:%M:%S")
435                    file_path = element[5]
436                except ValueError:
437                    date_and_time = datetime.strptime(
438                        element[5], "%d-%b-%Y %H:%M:%S")
439                    file_path = element[4]
440
441                meta_token = {
442                    'trial_number': element[0],
443                    'prompt': element[1],
444                    'date_and_time': date_and_time,
445                    'dat_filename': PureWindowsPath(file_path).name
446                }
447                meta.append(meta_token)
448        return meta
