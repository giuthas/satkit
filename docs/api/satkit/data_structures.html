 89class PointAnnotations():
 90    """
 91    Time point annotations for a Modality.
 92
 93    For each modality there should be only one of these for each kind of
 94    annotation type. 
 95
 96    annotation_type : AnnotationType
 97        unique identifier for the annotation type
 98    indeces : np.ndarray
 99        indeces of the annotation points. `modality_data.data[indeces[i]]` and
100        `modality_data.timevector[indeces[i]]` correspond to the annotation at
101        `i`.
102    times : np.ndarray 
103        timestamps of the annotation points
104    generating_parameters : dict 
105        the function call arguments and other parameters used in generating
106        these annotations.
107    properties : dict
108        a dictionary containing arrays of each of the annotation properties
109        expected for this annotation type.
110    """
111    annotation_type: AnnotationType
112    indeces: np.ndarray
113    times: np.ndarray
114    generating_parameters: dict
115    properties: dict

143    def apply_lower_time_limit(self, time_min: float) -> None:
144        """
145        Apply a lower time limit to the annotations.
146
147        This removes the annotation points before the given time limit.
148
149        Parameters
150        ----------
151        time_min : float
152            The time limit.
153        """
154        selected = np.nonzero(self.times >= time_min)
155        self.indeces = self.indeces[selected]
156        self.times = self.times[selected]
157        limit = selected[0]
158
159        for key in self.properties:
160            if is_sequence_form(self.properties[key]):
161                ic(key, self.properties[key])
162                self.properties[key] = self.properties[key][limit:]
163            elif isinstance(self.properties[key], np.ndarray):
164                self.properties[key] = self.properties[key][selected]
165
166    def apply_upper_time_limit(self, time_max: float) -> None:
167        """
168        Apply an upper time limit to the annotations.
169
170        This removes the annotation points after the given time limit.
171
172        Parameters
173        ----------
174        time_max : float
175            The time limit.
176        """
177        selected = np.nonzero(self.times <= time_max)
178        self.indeces = self.indeces[selected]
179        self.times = self.times[selected]
180        limit = selected[-1]
181
182        for key in self.properties:
183            if is_sequence_form(self.properties[key]):
184                ic(key, self.properties[key])
185                self.properties[key] = self.properties[key][:limit]
186            elif isinstance(self.properties[key], np.ndarray):
187                self.properties[key] = self.properties[key][selected]

233class Recording(UserDict):
234    """
235    A Recording is a dictionary of 0-n synchronised Modalities.
236
237    Adding modalities can be done by `recording[name] = modality`, but
238    `recording.add_modality` is preferred as a safer way which checks for
239    overwriting. The reason Recording is a dictionary is to make it possible to
240    iterate with the idiom `for modality_name in recording`.
241
242    The recording also contains the non-modality-specific metadata
243    (participant, speech content, etc) as a dictionary, as well as the textgrid
244    for the whole recording.
245
246    In general, inheriting should not be necessary, but if it is, inheriting
247    classes should call self._read_textgrid() after calling super.__init__()
248    (with correct arguments) and doing any updates to self.meta['textgrid']
249    that are necessary.
250    """
251
252    def __init__(self,
253                 meta_data: RecordingMetaData,
254                 excluded: bool = False,
255                 textgrid_path: Union[str, Path] = "") -> None:
256        """
257        Construct a mainly empty recording without modalities.
258
259        Modalities and annotations get added after constructions with their own
260        add_[modality or annotation] functions.
261
262        Parameters
263        ----------
264        meta_data : RecordingMetaData
265            Some of the contents of the meta data are available as properties.
266        excluded : bool, optional
267            _description_, by default False
268        textgrid_path : Union[str, Path], optional
269            _description_, by default ""
270        """
271        super().__init__()
272
273        self.excluded = excluded
274
275        self.meta_data = meta_data
276
277        self.textgrid_path = textgrid_path
278        if not self.textgrid_path:
279            self.textgrid_path = meta_data.path.joinpath(
280                meta_data.basename + ".TextGrid")
281        self.textgrid = self._read_textgrid()
282        self.satgrid = SatGrid(self.textgrid)
283
284        self.annotations = {}
285
286    @property
287    def modalities(self) -> dict[str, 'Modality']:
288        """
289        Dictionary of the modalities of this Recording.
290
291        Returns
292        -------
293        dict[str, Modality]
294            The dictionary of the Modalities.
295        """
296        return self.data
297
298    @property
299    def path(self) -> Path:
300        """Path to this recording."""
301        return self.meta_data.path
302
303    @path.setter
304    def path(self, path: Path) -> None:
305        self.meta_data.path = path
306
307    @property
308    def basename(self) -> str:
309        """Filename of this Recording without extensions."""
310        return self.meta_data.basename
311
312    @basename.setter
313    def basename(self, basename: str) -> None:
314        self.meta_data.basename = basename

316    def _read_textgrid(self) -> textgrids.TextGrid:
317        """
318        Read the textgrid specified in self.meta.
319
320        If file does not exist or reading fails, recovery is attempted 
321        by logging an error and creating an empty textgrid for this 
322        recording.
323        """
324        textgrid = None
325        if self.textgrid_path.is_file():
326            try:
327                textgrid = textgrids.TextGrid(self.textgrid_path)
328                _datastructures_logger.info("Read textgrid in %s.",
329                                            self.textgrid_path)
330            except Exception as exception:
331                _datastructures_logger.critical(
332                    "Could not read textgrid in %s.", self.textgrid_path)
333                _datastructures_logger.critical(
334                    "Failed with: %s.", str(exception))
335                _datastructures_logger.critical("Creating an empty textgrid "
336                                                + "instead.")
337                textgrid = textgrids.TextGrid()
338        else:
339            notice = 'Note: ' + str(self.textgrid_path) + " did not exist."
340            _datastructures_logger.warning(notice)
341            _datastructures_logger.warning("Creating an empty textgrid "
342                                           + "instead.")
343            textgrid = textgrids.TextGrid()
344        return textgrid
345
346    def identifier(self) -> str:
347        """
348        Generate a unique identifier for this Recording from metadata.
349
350        Returns
351        -------
352        str
353            prompt followed by time of recording.
354        """
355        return (f"{self.meta_data.prompt} "
356                f"{str(self.meta_data.time_of_recording)}")
357
358    def exclude(self) -> None:
359        """
360        Set self.excluded to True with a method.
361
362        This method exists to facilitate list comprehensions being used
363        for excluding recordings e.g. 
364        [recording.exclude() for recording in recordings if in some_list].
365        """
366        self.excluded = True
367
368    def write_textgrid(self, filepath: Path = None) -> None:
369        """
370        Save this recording's textgrid to file.
371
372        Keyword argument:
373        filepath -- string specifying the path and name of the 
374            file to be written. If filepath is not specified, this 
375            method will try to overwrite the textgrid specified in 
376            self.meta.
377
378            If filepath is specified, subsequent calls to this 
379            function will write into the new path rather than 
380            the original one.
381        """
382        try:
383            if filepath:
384                self.textgrid_path = filepath
385                self.textgrid.write(filepath)
386            else:
387                self.textgrid.write(self.textgrid_path)
388        except Exception as exception:
389            _datastructures_logger.critical(
390                "Could not write textgrid to %s.", str(self.textgrid_path))
391            _datastructures_logger.critical(
392                "TextGrid save failed with error: %s", str(exception))
393
394    def add_modality(
395            self, modality: 'Modality', replace: bool = False) -> None:
396        """
397        This method adds a new Modality object to the Recording.
398
399        Replacing a modality has to be specified otherwise if a
400        Modality with the same name already exists in this Recording
401        and the replace argument is not True, an Error is raised. 
402
403        Arguments:
404        modality -- object of type Modality to be added to 
405            this Recording.
406
407        Keyword arguments:
408        replace -- a boolean indicating if an existing Modality should
409            be replaced.
410        """
411        name = modality.name
412
413        if name in self.modalities and not replace:
414            ic(modality.metadata)
415            ic(self[modality.name].metadata)
416            raise ModalityError(
417                "A modality named " + name +
418                " already exists and replace flag was False.")
419
420        if replace:
421            self.modalities[name] = modality
422            _datastructures_logger.debug("Replaced modality %s.", name)
423        else:
424            self.modalities[name] = modality
425            _datastructures_logger.debug("Added new modality %s.", name)
426
427    def __str__(self) -> str:
428        """
429        Bare minimum string representation of the Recording.
430
431        Returns
432        -------
433        string
434            'Recording [basename]'
435        """
436        return f"Recording {self.basename}"

439class Modality(abc.ABC, OrderedDict):
440    """
441    Abstract superclass for all data Modality classes.
442
443    Any annotations associated with a Modality instance are accessible directly
444    by `modality[annotation_type]`, because a Modality is also a OrderedDict of
445    its Annotations.
446    """
447
448    @classmethod
449    @abc.abstractmethod
450    def generate_name(cls, params: ModalityMetaData) -> str:
451        pass
452
453    def __init__(self,
454                 recording: Recording,
455                 parsed_data: Optional[ModalityData] = None,
456                 metadata: Optional[ModalityMetaData] = None,
457                 data_path: Optional[Path] = None,
458                 meta_path: Optional[Path] = None,
459                 load_path: Optional[Path] = None,
460                 time_offset: Optional[float] = None,
461                 annotations: Optional[dict[AnnotationType,
462                                            PointAnnotations]] = None
463                 ) -> None:
464        """
465        Modality constructor.
466
467        Positional arguments:
468        recording -- the containing Recording.
469
470        Keyword arguments:
471        data_path -- path of the data file
472        load_path -- path of data when saved by SATKIT - both data and metadata
473        parent -- the Modality this one was derived from. None means this 
474            is an underived data Modality.
475        parsed_data -- ModalityData object containing waveform, sampling rate,
476            and either timevector and/or time_offset. 
477        parsed_data -- a ModalityData object containing parsed data 
478            that's been either read from file, loaded from file 
479            (previously saved by SATKIT), or calculated from another modality.
480            Providing a timevector 
481            overrides any time_offset value given, but in absence of a 
482            timevector the time_offset will be applied on reading the data 
483            from file. 
484        time_offset -- offset of this modality in relation to the Recordings
485            baseline - usually the audio track. This will be ignored if 
486            parsed_data exists and effectively overridden by 
487            parsed_data.timevector.
488        """
489        self.recording = recording
490        self.data_path = data_path
491        self._meta_path = meta_path  # self.meta_path is a property
492        self.load_path = load_path
493
494        self.metadata = metadata
495        if annotations:
496            self.update(annotations)
497        # else:
498        #     self.annotations = {}
499
500        if parsed_data:
501            self._modality_data = parsed_data
502        else:
503            self._modality_data = None
504
505        if (self._modality_data is None or
506                self._modality_data.timevector is None):
507            self._time_offset = time_offset
508        else:
509            self._time_offset = self._modality_data.timevector[0]
510
511        # This is a property which when set to True will also set
512        # parent.excluded to True.
513        self.excluded = False
514        self._time_precision = None
515
516    @property
517    def annotations(self) -> dict[AnnotationType, PointAnnotations]:
518        """
519        Property which is the annotations dictionary of this Modality.
520
521        This is currently the Modality itself.
522
523        Returns
524        -------
525        dict[AnnotationType, PointAnnotations]
526            The dictionary.
527        """
528        return self

530    def _get_data(self) -> ModalityData:
531        # TODO: Provide a way to force the data to be derived or loaded.
532        # this would be used when parent modality has updated in some way
533        if self.data_path:
534            return self._read_data()
535
536        if self.load_path:
537            return self._load_data()
538
539        if self.metadata.parent_name:
540            return self._derive_data()
541        else:
542            raise MissingDataError(
543                "Asked to get data but have no path and no parent Modality.\n"
544                + "Don't know how to solve this.")

547    # just write the fields? And if return a value, why not

555
556        This method is intended to rely on self.parent to provide the data
557        to be processed.
558        """

560            "This method should be overridden by inheriting classes.")

590    def add_annotations(self, annotations: PointAnnotations) -> None:
591        """
592        Add the PointAnnotations object to this Modality.
593
594        If there were previous annotations in this Modality with the same
595        AnnotationType, they will be overwritten.

597        To add single annotation points, use the add_annotation method in
598        PointAnnotations.

616    @property

618        """
619        Identity and possible parent data class.

621        This will be just the class name if this is a data Modality instance.
622        For derived Modalities the name will be of the form
623        '[own class name] on [data modality class name]'.

634    @property
635    def modality_data(self) -> ModalityData:
636        """
637        The data of this Modality as a NumPy array. 
638
639        The data refers to the actual data this modality represents
640        and for DerivedModality it is the result of running the 
641        modality's algorithm on the original data.
642
643        The dimensions of the array are in the 
644        order of [time, others]
645
646        If this modality is not preloaded, accessing this property will
647        cause data to be loaded on the fly _and_ saved in memory. To 
648        release the memory, assign None to this Modality's data.
649        """
650        if self._modality_data is None:
651            _datastructures_logger.debug(
652                "In Modality modality_data getter. "
653                "self._modality_data was None.")
654            self._set_data(self._get_data())
655        return self._modality_data

664        modality's algorithm on the original data.
665
666        The dimensions of the array are in the 
667        order of [time, others]

669        If this modality is not preloaded, accessing this property will

698    def _data_setter(self, data: np.ndarray) -> None:
699        """Set the data property from this class and subclasses."""
700        if self.data is not None and data is not None:
