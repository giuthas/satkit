 54class QtAnnotatorWindow(QMainWindow, Ui_MainWindow):
 55    """
 56    QtAnnotatorWindow provides the GUI basis for Annotator.
 57
 58    In the Model-view-presenter pattern this class is the View, current
 59    Annotator is the Presenter and a Recordings array is the Model. See
 60    https://en.wikipedia.org/wiki/Model%E2%80%93view%E2%80%93presenter
 61
 62    The idea is for QtAnnotatorWindow to only have minimal functionality. Adding
 63    display views, buttons and such is the job of Annotator. Annotator in turn
 64    may delegate these to other classes.
 65
 66    Reason for not implementing much anything in QtAnnotatorWindow is two-fold:
 67    1. Inheriting a class that inherits QMainWindow and Ui_MainWindow does not
 68    seem to work. 2. We will try to keep inheritance trees fairly shallow to
 69    make them easier to manage and maintain.
 70    """

 72    @staticmethod
 73    def line_xdirection_picker(line, mouse_event):
 74        """
 75        Find the nearest point in the x (time) direction from the mouseclick in
 76        data coordinates. Return index of selected point, x and y coordinates of
 77        the data at that point, and inaxes to enable originating subplot to be
 78        identified.
 79        """
 80        if mouse_event.xdata is None:
 81            return False, dict()
 82        xdata = line.get_xdata()
 83        ydata = line.get_ydata()
 84        distances = np.abs(xdata - mouse_event.xdata)
 85
 86        ind = np.argmin(distances)
 87        # if 1:
 88        pickx = np.take(xdata, ind)
 89        picky = np.take(ydata, ind)
 90        props = dict(ind=ind,
 91                     pickx=pickx,
 92                     picky=picky,
 93                     inaxes=mouse_event.inaxes)
 94        return True, props
 95        # else:
 96        #     return False, dict()

230    def draw_plots(self):
231        """
232        Updates title and graphs. Called by self.update().
233        """
234        self.ax1.set_title(self._get_title())
235        # self.ax1.axes.xaxis.set_ticklabels([])
236
237        audio = self.current.modalities['MonoAudio']
238        stimulus_onset = audio.meta['stimulus_onset']
239        wav = audio.data
240        wav_time = (audio.timevector - stimulus_onset)
241
242        pd = self.current.modalities['PD on RawUltrasound']
243        ultra_time = pd.timevector - stimulus_onset
244
245        # self.xlim = [ultra_time[0] - 0.05, ultra_time[-1]+0.05]
246        self.xlim = [-0.25, 1.0]
247
248        textgrid = self.current.textgrid
249
250        # self.pd_boundaries = plot_pd(
251        # self.ax1, pd.data['pd'],
252        # ultra_time, self.xlim, textgrid, stimulus_onset,
253        # picker=PdQtAnnotator.line_xdirection_picker)
254        self.pd_boundaries = plot_timeseries(
255            self.ax1, pd.data['pd'],
256            ultra_time, self.xlim, textgrid, stimulus_onset)
257        self.wav_boundaries = plot_wav(self.ax3, wav, wav_time, self.xlim,
258                                       textgrid, stimulus_onset)
259
260        if self.current.annotations['pdOnset'] > -1:
261            self.ax1.axvline(x=self.current.annotations['pdOnset'],
262                             linestyle=':', color="deepskyblue", lw=1)
263            self.ax3.axvline(x=self.current.annotations['pdOnset'],
264                             linestyle=':', color="deepskyblue", lw=1)
265        if self.display_tongue:
266            self.draw_ultra_frame()
267
268    def draw_ultra_frame(self):
269        """
270        Display an already interpolated ultrasound frame.
271        """
272        index = 1
273        if self.current.annotations['pdOnsetIndex']:
274            index = self.current.annotations['pdOnsetIndex']
275        image = self.current.modalities['RawUltrasound'].interpolated_image(
276            index)
277        self.ultra_axes.imshow(image, interpolation='nearest', cmap='gray')
278
279    def draw_raw_ultra_frame(self):
280        """
281        Interpolate and display a raw ultrasound frame.
282        """
283        if self.current.annotations['pdOnsetIndex']:
284            ind = self.current.annotations['pdOnsetIndex']
285            array = self.current.modalities['RawUltrasound'].data[ind, :, :]
286        else:
287            array = self.current.modalities['RawUltrasound'].data[1, :, :]
288        array = np.transpose(array)
289        array = np.flip(array, 0).copy()
290        array = array.astype(np.int8)
291        self.ultra_axes.imshow(array, interpolation='nearest', cmap='gray')
