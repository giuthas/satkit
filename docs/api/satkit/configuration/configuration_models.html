These are the code friendly wrappers for the configuration structures.

configuration_parsers contains the actual parsing of strictyaml into comment-retaining dictionary-like structures. Here those structures get parsed into pydantic models that know what their fields actually are.

This two level structure is maintained so that in some version after SATKIT 1.0 we can implement configuration round tripping with preserved comments.

136class SearchPattern(UpdatableBaseModel):
137    """
138    Representation for simple and regexp search patterns.
139
140    Members
141    ----------
142    pattern : str
143        The pattern to search for
144    is_regexp : bool, optional
145        If the pattern should be treated as a regexp or not. Defaults to False.
146    """
147    pattern: str
148    is_regexp: bool = False
149
150    def match(self, string: str) -> bool:
151        """
152        Match this pattern to the argument string.
153
154        If this pattern is not a regexp then this method will return True only
155        when the pattern is found verbatim in the argument string.
156
157        Parameters
158        ----------
159        string : str
160            The string to match to.
161
162        Returns
163        -------
164        bool
165            True if this pattern matches the argument.
166        """
167        if self.is_regexp:
168            return bool(re.match(self.pattern, string))
169
170        return self.pattern in string
171
172    @staticmethod
173    def build(value: dict | str) -> 'SearchPattern':
174        """
175        Build a SearchPattern from a dictionary or a string.
176
177        Parameters
178        ----------
179        value : dict | str
180            The dictionary or string to build the SearchPattern from. If the
181            parameter is a string it is used as the pattern, if it is a dict
182            it's passed as keyword arguments to the SearchPattern constructor.
183
184        Returns
185        -------
186        SearchPattern
187            The constructed SearchPattern.
188
189        Raises
190        ------
191        ValueError
192            If the value is not a dict or a string.
193        """
194        if isinstance(value, str):
195            return SearchPattern(pattern=value)
196
197        if isinstance(value, dict):
198            return SearchPattern(**value)
199
200        raise ValueError("Unrecognised input value type: "
201                         + type(value).__name__ + ".")

204class TimeseriesNormalisation(UpdatableBaseModel):
205    """
206    Selection between peak normalised, bottom normalised or both.
207
208    Contains a boolean for each peak and bottom normalisation.
209    """
210    peak: bool = False
211    bottom: bool = False
212
213    # TODO: this class needs a special dumper to save things correctly.
214
215    @staticmethod
216    def build(
217            value: str
218    ) -> 'TimeseriesNormalisation':
219        """
220        Construct a TimeseriesNormalisation object from a string value.
221
222        The value usually comes from a config file.
223
224        Parameters
225        ----------
226        value : str
227            'none': no normalisation
228            'peak': divide all data points y-values by the largest y-value
229            'bottom': deduct the lowest y-value from all data points y-values
230            'both': do first bottom normalisation and then peak normalisation.
231
232        Returns
233        -------
234        TimeseriesNormalisation
235            The new TimeseriesNormalisation with fields set as expected.
236        """
237        match value:
238            case 'none':
239                return TimeseriesNormalisation()
240            case 'peak':
241                return TimeseriesNormalisation(peak=True)
242            case 'bottom':
243                return TimeseriesNormalisation(bottom=True)
244            case 'both':
245                return TimeseriesNormalisation(peak=True, bottom=True)
246            case _:
247                raise ValueError("Unrecognised value: " + value + ".")
