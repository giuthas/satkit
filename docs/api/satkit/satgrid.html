 78class SatPoint(SatAnnotation):
 79    """TextGrid Point representation to enable editing with GUI."""
 80
 81    @classmethod
 82    def from_textgrid_point(
 83            cls,
 84            point: Point,
 85    ) -> Self:
 86        """
 87        Copy the info of a Python TextGrids Interval into a new SatInterval.
 88
 89        Only xmin and text are copied from the original Interval. xmax is
 90        assumed to be handled by either the next SatInterval or the
 91        constructing method if this is the last Interval.
 92
 93        Since SatIntervals are doubly linked, an attempt will be made to link
 94        prev and next to this interval.
 95
 96        Returns the newly created SatInterval.
 97        """
 98        return cls(
 99            time=point.xpos,
100            label=point.text,
101        )

106            label: None | Transcript
107    ) -> None:

112        """Location of this Point."""
113        return self._time

116    def time(self, time: float) -> None:
117        self._time = time
118
119    def contains(self, time: float) -> bool:
120        epsilon = config_dict['epsilon']
121        if self._time - epsilon < time < self._time + epsilon:
122            return True
123        return False

126class SatInterval(SatAnnotation):
127    """TextGrid Interval representation to enable editing with GUI."""
128
129    @classmethod
130    def from_textgrid_interval(
131        cls,
132        interval: Interval,
133        prev_interval: None | Self,
134        next_interval: None | Self = None
135    ) -> Self:
136        """
137        Copy the info of a Python TextGrids Interval into a new SatInterval.
138
139        Only xmin and text are copied from the original Interval. xmax is
140        assumed to be handled by either the next SatInterval or the
141        constructing method if this is the last Interval. 
142
143        Since SatIntervals are doubly linked, an attempt will be made to link
144        prev and next to this interval. 
145
146        Returns the newly created SatInterval.
147        """
148        return cls(
149            begin=interval.xmin,
150            label=interval.text,
151            prev_interval=prev_interval,
152            next_interval=next_interval)
153
154    def __init__(self,
155                 begin: float,
156                 label: None | Transcript,
157                 prev_interval: None | Self = None,
158                 next_interval: None | Self = None) -> None:
159        super().__init__(
160            time=begin,
161            label=label,
162        )
163
164        self._prev_interval = prev_interval
165        if self.prev:
166            self.prev._next_interval = self
167
168        self._next_interval = next_interval
169        if self.next:
170            self.next.prev = self
171
172    def __repr__(self) -> str:
173        return (f"{self.__class__.__name__}: text: '{self.label}'\t "
174                f"begin: {self.begin}, end: {self.end}")
175
176    @property
177    def next(self) -> Self | None:
178        """The next annotation, if any."""
179        return self._next_interval
180
181    @next.setter
182    def next(self, next_interval: Self | None) -> None:
183        if next_interval != self._next_interval:
184            self._next_interval = next_interval
185
186    @property
187    def prev(self) -> Self | None:
188        """The previous annotation, if any."""
189        return self._prev_interval
190
191    @prev.setter
192    def prev(self, prev_interval: Self | None) -> None:
193        if prev_interval != self._prev_interval:
194            self._prev_interval = prev_interval
195
196    @property
197    def begin(self) -> float:
198        """Beginning time point of the interval."""
199        return self._time
200
201    @begin.setter
202    def begin(self, value: float) -> None:
203        self._time = value
204
205    @property
206    def mid(self) -> float | None:
207        """
208        Middle time point of the interval.
209
210        This is a property that will return None
211        if this Interval is the one that marks
212        the last boundary.
213        """
214        if self._next_interval:
215            return (self.begin + self._next_interval.begin)/2
216        return None
217
218    @property
219    def end(self) -> float | None:
220        """
221        End time point of the interval.
222
223        This is a property that will return None
224        if this Interval is the one that marks
225        the last boundary.
226        """
227        if self._next_interval:
228            return self._next_interval.begin
229        return None
230
231    @end.setter
232    def end(self, value: float) -> None:
233        if self._next_interval:
234            self._next_interval.begin = value
235
236    def is_at_time(self, time: float, epsilon) -> bool:
237        """
238        Intervals are considered equivalent if the difference between their
239        `begin` values is < epsilon. Epsilon is a constant defined in SATKIT's
240        configuration.
241        """
242        return abs(self.begin - time) < epsilon
243
244    def is_last(self) -> bool:
245        """Is this the last Interval in this Tier."""
246        return self._next_interval is None
247
248    def is_legal_value(self, time: float, epsilon: float) -> bool:
249        """
250        Check if the given time is between the previous and next boundary.
251
252        Usual caveats about float testing don't apply, because each boundary is
253        padded with SATKIT epsilon. Tests used do not include equality with
254        either bounding boundary, and that may or may not be trusted to be the
255        actual case depending on how small the epsilon is.
256
257        Returns True, if time is  between the previous and next boundary.
258        """
259        return (time + epsilon < self._next_interval.begin and
260                time > epsilon + self.prev.begin)
261
262    def contains(self, time: float) -> bool:
263        if self.begin < time < self.end:
264            return True
265        return False

268class SatTier(list):
269    """TextGrid Tier representation to enable editing with GUI."""
270
271    @classmethod
272    def from_textgrid_tier(cls, tier: Tier) -> Self:
273        """
274        Copy a Python TextGrids Tier as a SatTier.
275
276        Returns the newly created SatTier.
277        """
278        return cls(tier)
279
280    def __init__(self, tier: Tier) -> None:
281        super().__init__()
282        last_interval = None
283        prev = None
284        for interval in tier:
285            current = SatInterval.from_textgrid_interval(interval, prev)
286            self.append(current)
287            prev = current
288            last_interval = interval
289        self.append(SatInterval(last_interval.xmax, None, prev))
290
291    def __repr__(self) -> str:
292        representation = f"{self.__class__.__name__}:\n"
293        for interval in self:
294            representation += str(interval) + "\n"
295        return representation
296
297    @property
298    def begin(self) -> float:
299        """
300        Begin timestamp.
301
302        Corresponds to a TextGrid Interval's xmin.
303
304        This is a property and the actual value is generated from the first
305        SatInterval of this SatTier.
306        """
307        return self[0].begin
308
309    @property
310    def end(self) -> float:
311        """
312        End timestamp.
313
314        Corresponds to a TextGrid Interval's xmin.
315
316        This is a property and the actual value is generated from the last
317        SatInterval of this SatTier.
318        """
319        # This is slightly counterintuitive, but the last interval is in fact
320        # empty and only represents the final boundary. So its `begin` is the
321        # final boundary.
322        return self[-1].begin
323
324    @property
325    def is_point_tier(self) -> bool:
326        """Is this Tier a PointTier."""
327        return False
328
329    def boundary_at_time(
330            self, time: float, epsilon: float) -> SatInterval | None:
331        """
332        If there is a boundary at time, return it.
333
334        Returns None, if there is no boundary at time. 
335
336        'Being at time' is defined as being within SATKIT epsilon of the given
337        timestamp.
338        """
339        for interval in self:
340            if interval.is_at_time(time=time, epsilon=epsilon):
341                return interval
342        return None
343
344    def get_interval_by_category(
345        self,
346        interval_category: IntervalCategory,
347        label: str | None = None
348    ) -> SatInterval:
349        """
350        Return the Interval matching the category in this Tier.
351
352        If interval_category is FIRST_LABELED or LAST_LABELED, the label should
353        be specified as well.
354
355        Parameters
356        ----------
357        interval_category : IntervalCategory
358            The category to search for.
359        label : Optional[str], optional
360            Label to search for when doing a label based category search, by
361            default None
362
363        Returns
364        -------
365        SatInterval
366            _description_
367        """
368        if interval_category is IntervalCategory.FIRST_NON_EMPTY:
369            for interval in self:
370                if interval.label:
371                    return interval
372
373        if interval_category is IntervalCategory.LAST_NON_EMPTY:
374            for interval in reversed(self):
375                if interval.label:
376                    return interval
377
378        if interval_category is IntervalCategory.FIRST_LABELED:
379            for interval in self:
380                if interval.label == label:
381                    return interval
382
383        if interval_category is IntervalCategory.LAST_LABELED:
384            for interval in reversed(self):
385                if interval.label == label:
386                    return interval
387
388    def get_labels(self, time_vector: np.ndarray) -> np.ndarray:
389        """
390        Get the labels at the times in the `time_vector`.
391
392        Parameters
393        ----------
394        time_vector : np.ndarray
395            Time stamps to retrieve the labels for.
396
397        Returns
398        -------
399        np.ndarray
400            This array contains the labels as little endian Unicode strings.
401        """
402        max_label = max(
403            [len(element.label) for element in self
404             if element.label is not None]
405        )
406        labels = np.empty(len(time_vector), dtype=f"<U{max_label}")
407        for (i, time) in enumerate(time_vector):
408            labels[i] = self.label_at(time)
409        return labels
410
411    def label_at(self, time: float) -> str:
412        """
413        Get the label at the given time.
414
415        Parameters
416        ----------
417        time : float
418            Time in seconds to retrieve the label for.
419
420        Returns
421        -------
422        str
423            The label string.
424        """
425        if time < self.begin or time > self.end:
426            return ""
427
428        for element in self:
429            if element.contains(time):
430                return element.label

433class SatGrid(OrderedDict):
434    """
435    TextGrid representation which makes editing easier.
436
437    SatGrid is a OrderedDict very similar to Python textgrids TextGrid, but
438    made up of SatTiers that in turn contain intervals or points as doubly
439    linked lists instead of just lists. See the relevant classes for more
440    details.
441    """
442
443    def __init__(self, textgrid: TextGrid) -> None:
444        super().__init__()
445        for tier_name in textgrid:
446            self[tier_name] = SatTier.from_textgrid_tier(textgrid[tier_name])
447
448    # def as_textgrid(self):
449    #     pass
450
451    @property
452    def begin(self) -> float:
453        """
454        Begin timestamp.
455
456        Corresponds to a TextGrids xmin.
457
458        This is a property and the actual value is generated from the first
459        SatTier of this SatGrid.
460        """
461        key = list(self.keys())[0]
462        return self[key].begin
463
464    @property
465    def end(self) -> float:
466        """
467        End timestamp.
468
469        Corresponds to a TextGrids xmax.
470
471        This is a property and the actual value is generated from the first
472        SatTier of this SatGrid.
473        """
474        # First Tier
475        key = list(self.keys())[0]
476        # Return the end of the first Tier.
477        return self[key].end
478
479    def format_long(self) -> str:
480        """Format self as long format TextGrid."""
481        out = long_header.format(self.begin, self.end, len(self))
482        tier_count = 1
483        for name, tier in self.items():
484            if tier.is_point_tier:
485                tier_type = 'PointTier'
486                elem_type = 'points'
487            else:
488                tier_type = 'IntervalTier'
489                elem_type = 'intervals'
490            out += long_tier.format(tier_count,
491                                    tier_type,
492                                    name,
493                                    self.begin,
494                                    self.end,
495                                    elem_type,
496                                    len(tier)-1)
497            for elem_count, elem in enumerate(tier, 1):
498                if tier.is_point_tier:
499                    out += long_point.format(elem_count,
500                                             elem.time,
501                                             elem.label)
502                elif elem.next:
503                    out += long_interval.format(elem_count,
504                                                elem.begin,
505                                                elem.end,
506                                                elem.label)
507                else:
508                    # The last interval does not contain anything.
509                    # It only marks the end of the file and final
510                    # interval's end. That info got already used by
511                    # elem.end (which is elem.next.begin) above.
512                    pass
513        return out
514
515    def get_labels(self, time_vector: np.ndarray) -> dict[str, np.ndarray]:
516        """
517        Get the labels at the times in the `time_vector`.
518
519        Parameters
520        ----------
521        time_vector : np.ndarray
522            Time values to get the labels for.
523
524        Returns
525        -------
526        dict[str, np.ndarray]
527            Dictionary of the labels indexed by tier name.
528        """
529        labels = {}
530        for tier_name in self:
531            labels[tier_name] = self[tier_name].get_labels(time_vector)
532
533        return labels
