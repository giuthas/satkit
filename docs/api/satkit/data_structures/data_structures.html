 61class Session(DataAggregator, UserList):
 62    """
 63    The metadata and Recordings of a recording session.
 64
 65    This class behaves exactly like a list of Recordings with some extra
 66    fields. While some legacy code may be left behind, the preferred idiom for
 67    iterating over the recordings is `for recording in recording_session:`.
 68
 69    Sessions can also hold aggregate data in the form of Statistics.
 70    """
 71
 72    def __init__(
 73            self,
 74            name: str,
 75            paths: PathStructure,
 76            config: SessionConfig,
 77            file_info: FileInformation,
 78            recordings: list[Recording],
 79            statistics: dict[str, Statistic] | None = None
 80    ) -> None:
 81        super().__init__(
 82            owner=None, name=name, metadata=config,
 83            file_info=file_info, statistics=statistics)
 84
 85        for recording in recordings:
 86            if not recording.owner:
 87                recording.owner = self
 88        self.extend(recordings)
 89
 90        self.paths = paths
 91        # This was commented out in data structures 4.0, and left here to
 92        # explain what session.config was. Remove if nothing broke.
 93        # self.config = config
 94
 95    @property
 96    def recordings(self) -> list[Recording]:
 97        """
 98        Property to access the list of Recordings directly.
 99
100        Returns
101        -------
102        list[Recording]
103            The list of this Session's Recordings.
104        """
105        return self.data

108class Recording(DataAggregator, UserDict):
109    """
110    A Recording is a dictionary of 0-n synchronised Modalities.
111
112    Adding modalities can be done by `recording[name] = modality`, but
113    `recording.add_modality` is preferred as a safer way which checks for
114    overwriting. The reason Recording is a dictionary is to make it possible to
115    iterate with the idiom `for modality_name in recording`.
116
117    The recording also contains the non-modality-specific metadata
118    (participant, speech content, etc.) as a dictionary, as well as the textgrid
119    for the whole recording.
120
121    In general, inheriting should not be necessary, but if it is, inheriting
122    classes should call `self._read_textgrid()` after calling `super.__init__()`
123    (with correct arguments) and doing any updates to `self.meta['textgrid']`
124    that are necessary.
125    """
126    owner: Session
127
128    def __init__(
129            self,
130            metadata: RecordingMetaData,
131            file_info: FileInformation,
132            owner: Session | None = None,
133            excluded: bool = False,
134            textgrid_path: str | Path = ""
135    ) -> None:
136        """
137        Construct a mainly empty recording without modalities.
138
139        Modalities and annotations get added after constructions with their own
140        add_[modality or annotation] functions.
141
142        NOTE: `after_modalities_init` should be called on each new Recording
143        after modalities have been loaded. It ensures that there is at least a
144        minimal TextGrid in place to facilitate GUI functions.
145
146        Parameters
147        ----------
148        metadata : RecordingMetaData
149            Some of the contents of the meta data are available as properties.
150        excluded : bool, optional
151            _description_, by default False
152        textgrid_path : Union[str, Path], optional
153            _description_, by default ""
154        """
155        super().__init__(
156            owner=owner, name=metadata.basename,
157            metadata=metadata, file_info=file_info)
158
159        self.excluded = excluded
160
161        self.textgrid_path = textgrid_path
162        if not self.textgrid_path:
163            self.textgrid_path = metadata.path.joinpath(
164                metadata.basename + ".TextGrid")
165        self.textgrid = self._read_textgrid()
166        if self.textgrid:
167            self.satgrid = SatGrid(self.textgrid)
168        else:
169            self.satgrid = None
170
171        self.annotations = {}
172

203    def _read_textgrid(self) -> textgrids.TextGrid | None:
204        """
205        Read the textgrid specified in self.meta.
206
207        If file does not exist or reading fails, recovery is attempted 
208        by logging an error and creating an empty textgrid for this 
209        recording.
210        """
211        textgrid = None
212        if self.textgrid_path.is_file():
213            try:
214                textgrid = textgrids.TextGrid(self.textgrid_path)
215                _logger.info("Read textgrid in %s.",
216                             self.textgrid_path)
217            except Exception as exception:
218                _logger.critical(
219                    "Could not read textgrid in %s.", self.textgrid_path)
220                _logger.critical(
221                    "Failed with: %s.", str(exception))
222        else:
223            notice = 'Note: ' + str(self.textgrid_path) + " did not exist."
224            _logger.warning(notice)
225        return textgrid

227    def identifier(self) -> str:
228        """
229        Generate a unique identifier for this Recording from metadata.
230
231        Returns
232        -------
233        str
234            prompt followed by time of recording.
235        """
236        return (f"{self.metadata.prompt} "
237                f"{str(self.metadata.time_of_recording)}")

239    def exclude(self) -> None:
240        """
241        Set `self.excluded` to True with a method.
242
243        This method exists to facilitate list comprehensions being used
244        for excluding recordings e.g. 
245        [recording.exclude() for recording in recordings if in some_list].
246        """
247        self.excluded = True
248
249    def write_textgrid(self, filepath: Path = None) -> None:
250        """
251        Save this recording's textgrid to file.
252
253        Keyword argument:
254        filepath -- string specifying the path and name of the 
255            file to be written. If filepath is not specified, this 
256            method will try to overwrite the textgrid specified in 
257            self.meta.
258
259            If filepath is specified, subsequent calls to this 
260            function will write into the new path rather than 
261            the original one.
262        """
263        try:
264            if filepath:
265                self.textgrid_path = filepath
266                self.textgrid.write(filepath)
267            else:
268                self.textgrid.write(self.textgrid_path)
269        except Exception as exception:
270            _logger.critical(
271                "Could not write textgrid to %s.", str(self.textgrid_path))
272            _logger.critical(
273                "TextGrid save failed with error: %s", str(exception))
274
275    def add_modality(
276            self, modality: Modality, replace: bool = False
277    ) -> None:
278        """
279        This method adds a new Modality object to the Recording.
280
281        Replacing a modality has to be specified otherwise if a
282        Modality with the same name already exists in this Recording
283        and the `replace` argument is not True, an Error is raised.
284
285        Arguments:
286        modality -- object of type Modality to be added to 
287            this Recording.
288
289        Keyword arguments:
290        replace -- a boolean indicating if an existing Modality should
291            be replaced.
292        """
293        name = modality.name
294
295        if name in self.modalities and not replace:
296            # ic(modality.metadata)
297            # ic(self[modality.name].metadata)
298            raise OverwriteError(
299                "A modality named " + name +
300                " already exists and replace flag was False.")
301
302        if replace:
303            self.modalities[name] = modality
304            _logger.debug("Replaced modality %s.", name)
305        else:
306            self.modalities[name] = modality
307            _logger.debug("Added new modality %s.", name)
308
309    def after_modalities_init(self) -> None:
310        """
311        Ensure everything is properly in place after loading modalities.
312
313        Currently, this is only used to create placeholder TextGrids when
314        needed.
315        """
316        if self.textgrid is None:
317            if 'MonoAudio' in self.modalities:
318                _logger.warning(
319                    "%s: Creating a placeholder textgrid to make GUI function "
320                    "correctly.", self.basename)
321                textgrid = textgrids.TextGrid()
322                interval = textgrids.Interval(
323                    text=self.metadata.prompt,
324                    xmin=self['MonoAudio'].min_time,
325                    xmax=self['MonoAudio'].max_time,
326                )
327                tier = textgrids.Tier(
328                    xmin=self['MonoAudio'].min_time,
329                    xmax=self['MonoAudio'].max_time,
330                )
331                tier.append(interval)
332                textgrid['Utterance'] = tier
333                self.textgrid = textgrid
334                self.satgrid = SatGrid(self.textgrid)
335            else:
336                _logger.warning("No audio found for %s.",
337                                self.basename)
338                _logger.warning("Can't create a textgrid so GUI may not "
339                                "function correctly.")
340
341    def __str__(self) -> str:
342        """
343        Bare minimum string representation of the Recording.
344
345        Returns
346        -------
347        string
348            'Recording [basename]'
349        """
350        return f"Recording {self.name}"
351
352    def __repr__(self) -> str:
353        """Overrides the default implementation"""
354        modalities = "{\n"
355        for modality in self.modalities:
356            modalities += ("\t" + modality
357                           + f": {self.modalities[modality].name},\n")
358        modalities += "}"
359        return (
360            f"Recording {self.basename}\n"
361            f"{modalities}")
362
363    def __eq__(self, other):
364        """Overrides the default implementation"""
365        if isinstance(other, Recording):
366            return self.name == other.name
367        return NotImplemented

370class Modality(DataContainer, OrderedDict):
371    """
372    Abstract superclass for all data Modality classes.
373
374    Any annotations associated with a Modality instance are accessible directly
375    by `modality[annotation_type]`, because a Modality is also a OrderedDict of
376    its Annotations.
377    """
378    owner: Recording
379
380    @classmethod
381    @abc.abstractmethod
382    def generate_name(cls, params: ModalityMetaData) -> str:
383        """Abstract version of generating a Modality name."""
384
385    def __init__(
386            self,
387            owner: Recording,
388            file_info: FileInformation,
389            parsed_data: ModalityData | None = None,
390            metadata: ModalityMetaData | None = None,
391            time_offset: float | None = None,
392            point_annotations: dict[AnnotationType, PointAnnotations] | None =
393            None
394    ) -> None:
395        """
396        Modality constructor.
397
398        Positional arguments:
399        recording -- the containing Recording.
400
401        Keyword arguments:
402        data_path -- path of the data file
403        load_path -- path of data when saved by SATKIT - both data and metadata
404        parent -- the Modality this one was derived from. None means this 
405            is a recorded data Modality.
406        parsed_data -- ModalityData object containing waveform, sampling rate,
407            and either timevector and/or time_offset. 
408        parsed_data -- a ModalityData object containing parsed data 
409            that's been either read from file, loaded from file 
410            (previously saved by SATKIT), or calculated from another modality.
411            Providing a timevector 
412            overrides any time_offset value given, but in absence of a 
413            timevector the time_offset will be applied on reading the data 
414            from file. 
415        time_offset -- offset of this modality in relation to the Recordings
416            baseline - usually the audio track. This will be ignored if 
417            parsed_data exists and effectively overridden by 
418            parsed_data.timevector.
419        """
420        super().__init__(
421            owner=owner,
422            metadata=metadata,
423            file_info=file_info)
424
425        if point_annotations:
426            self.update(point_annotations)
427
428        if parsed_data:
429            self._modality_data = parsed_data
430        else:
431            self._modality_data = None
432
433        if (self._modality_data is None or
434                self._modality_data.timevector is None):
435            self._time_offset = time_offset
436        else:
437            self._time_offset = self._modality_data.timevector[0]
438
439        # This is a property which when set to True will also set
440        # parent.excluded to True.
441        self.excluded = False
442        self._time_precision = None

557    @property

583        The data of this Modality as a NumPy array. 
584

603    @data.setter
604    def data(self, data: np.ndarray) -> None:
605        """
606        Data setter method.
607
608        Arguments: 
609        data -- either None or a numpy.ndarray with same dtype, size, 
610            and shape as self.data.
611
612        Assigning anything but None or a numpy ndarray with matching dtype,
613        size, and shape will raise a OverWriteError.
614
615        If shape of the data were to change then also shape of the timevector
616        should potentially change. Unlikely that we'd try to deal that in any
617        other way but to create a new Modality or even Recording.
618        """
619        self._data_setter(data)

621    def _data_setter(self, data: np.ndarray) -> None:
622        """Set the data property from this class and subclasses."""
623        if self.data is not None and data is not None:
624            if (data.dtype == self._modality_data.data.dtype and
625                    data.size == self._modality_data.data.size and
626                    data.shape == self._modality_data.data.shape):
627                self._modality_data.data = data
628            else:
629                raise DimensionMismatchError(
630                    "Trying to write over raw ultrasound data with a numpy " +
631                    "array that has non-matching dtype, size, or shape.\n" +
632                    " data.shape = " + str(data.shape) + "\n" +
633                    " self.data.shape = " +
634                    str(self._modality_data.data.shape) + ".")
635        else:
636            self._modality_data.data = data

650    @property
651    def time_offset(self):
652        """
653        The time offset of this modality.
654
655        Assigning a value to this property is implemented so that
656        `self.timevector[0]` stays equal to `self._time_offset`.
657
658        If shape of the timevector were to change then also shape of the data
659        should change. Unlikely that we'd try to deal that in any other way but
660        to create a new Modality or even Recording.
661        """
662        if not self._time_offset:
663            self._set_data(self._get_data())
664        return self._time_offset

666    @property
667    def time_precision(self) -> float:
668        """
669        Timevector precision: the maximum of absolute deviations.
670
671        Essentially this means that we are guesstimating the timevector to be
672        no more precise than the largest deviation from the average timestep.
673        """
674        if not self._time_precision:
675            differences = np.diff(self.timevector)
676            average = np.average(differences)
677            deviations = np.abs(differences - average)
678            self._time_precision = np.max(deviations)
679        return self._time_precision
680
681    @time_offset.setter
682    def time_offset(self, time_offset):
683        self._time_offset = time_offset
684        if self._modality_data.timevector:
685            self._modality_data.timevector = (
686                    self.timevector + (time_offset - self.timevector[0]))

688    @property
689    def timevector(self):
690        """
691        The timevector corresponding to `self.data` as a NumPy array.
692
693        If the data has not been previously loaded, accessing this 
694        property will cause data to be loaded on the fly _and_ saved 
695        in memory. To release the memory, assign None to this 
696        Modality's data. If the data has been previously 
697        loaded and after that released, the timevector still persists and  
698        accessing it does not trigger a new loading operation.
699
700        Assigning a value to this property is implemented so 
701        that `self.timevector[0]` stays equal to `self._timeOffset`.
702        """
703        if (self._modality_data is None or
704                self._modality_data.timevector is None):
705            self._set_data(self._get_data())
706        return self._modality_data.timevector
707
708    @timevector.setter
709    def timevector(self, timevector):
710        if (self._modality_data is None or
711                self._modality_data.timevector is None):
712            raise DimensionMismatchError(
713                "Trying to overwrite the timevector when "
