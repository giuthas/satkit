 90class SplineMetricParameters(ModalityMetaData):
 91    """
 92    Parameters used in generating a SplineMetric modality.
 93
 94    Parameters
 95    ----------
 96    metric : SplineMetricEnum
 97        A string specifying this Modality's metric. Defaults to the l1 norm.
 98    timestep : PositiveInt
 99        A  positive integer used as the timestep in calculating this Modality's
100        data. Defaults to 1, which means comparison of consecutive frames.
101    release_data_memory : bool
102        Whether to assign None to `parent.data` after deriving this Modality
103        from the data. Currently, has no effect as deriving SplineMetric at
104        runtime is not yet supported.
105    exclude_points : tuple[int, int]
106        How many points to exclude from the (beginning, end) of the spline,
107        by default, None.
108    """
109    parent_name: str
110    metric: SplineMetricEnum = SplineDiffsEnum.MPBPD
111    timestep: PositiveInt = 1
112    release_data_memory: bool = False
113    exclude_points: Optional[tuple[int, int]] = None

116class SplineMetric(Modality):
117    """
118    Represent a SplineMetric as a Modality. 
119    """
120
121    @classmethod
122    def generate_name(cls, params: SplineMetricParameters) -> str:
123        """
124        Generate a SplineMetric name to be used as its unique identifier.
125
126        This static method **defines** what the names are. This implementation
127        pattern (SplineMetric.name calls this and anywhere that needs to guess
128        what a name would be calls this) is how all derived Modalities should
129        work.
130
131        Parameters
132        ----------
133        params : SplineMetricParameters
134            The parameters of the SplineMetric instance. Note that this
135            SplineMetricParameters instance does not need to be attached to a
136            SplineMetric instance.
137
138        Returns
139        -------
140        str
141            Name of the SplineMetric instance.
142        """
143        name_string = cls.__name__ + " " + params.metric.value
144
145        if params.timestep != 1 and params.metric not in SplineShapesEnum:
146            name_string = name_string + " ts" + str(params.timestep)
147
148        if params.parent_name:
149            name_string = name_string + " on " + params.parent_name
150
151        if params.is_downsampled:
152            name_string += " downsampled by " + params.downsampling_ratio
153
154        return name_string
155
156    @staticmethod
157    def get_names_and_meta(
158        modality: Modality,
159        metrics: list[str] = None,
160        timesteps: list[int] = None,
161        exclude_points: tuple[int, int] = None,
162        release_data_memory: bool = False
163    ) -> dict[str: SplineMetricParameters]:
164        """
165        Generate SplineMetric modality names and metadata.
166
167        This method will generate the full cartesian product of the possible
168        combinations. If only some of them are needed, make more than one call
169        or weed the results afterwards.
170
171        Parameters
172        ----------
173        modality : Modality
174            parent modality that SplineMetric would be derived from
175        metrics : List[str], optional
176            list of metrics to be calculated, defaults to 'l2'.
177        timesteps : List[int], optional
178            list of timesteps to be used, defaults to 1.
179        exclude_points : tuple[int, int]
180            How many points to exclude from the (beginning, end) of the spline,
181            by default, None.
182        release_data_memory: bool
183            Should parent Modality's data be assigned to None after calculations
184            are complete, by default False.
185
186        Returns
187        -------
188        dict[str: SplineMetricParameters]
189            Dictionary where the names of the SplineMetric Modalities index the
190            SplineMetricParameter objects.
191        """
192        parent_name = modality.__name__
193
194        if not metrics:
195            metrics = SplineDiffsEnum.MPBPD
196        if not timesteps:
197            timesteps = [1]
198
199        param_dict = {
200            'parent_name': [parent_name],
201            'metric': metrics,
202            'timestep': timesteps,
203            'exclude_points': [exclude_points],
204            'release_data_memory': [release_data_memory]}
205
206        spline_metric_params = [SplineMetricParameters(**item)
207                                for item in product_dict(**param_dict)]
208
209        return {SplineMetric.generate_name(params): params
210                for params in spline_metric_params}
211
212    def __init__(self,
213                 owner: Recording,
214                 metadata: SplineMetricParameters,
215                 file_info: FileInformation,
216                 parsed_data: Optional[ModalityData] = None,
217                 time_offset: Optional[float] = None) -> None:
218        """
219        Build a SplineMetric Modality.       
220
221        Parameters
222        ----------
223        owner : Recording
224            the containing Recording.
225        metadata : SplineMetricParameters
226            Parameters used in calculating this instance of SplineMetric.
227        file_info : FileInformation
228            _description_
229        parsed_data : Optional[ModalityData], optional
230            ModalityData object, by default None. Contains SplineMetric values,
231            sampling rate,and either timevector and/or time_offset. Providing a
232            timevector overrides any time_offset value given, but in absence of
233            a timevector the time_offset will be applied on reading the data
234            from file.
235        time_offset : Optional[float], optional
236            Time offset in seconds against the Recordings baseline. If not
237            specified or 0, timeOffset will be copied from dataModality, by
238            default None
239        """
240        # This allows the caller to be lazy.
241        if not time_offset:
242            if parsed_data:
243                time_offset = parsed_data.timevector[0]
244
245        super().__init__(
246            owner=owner,
247            metadata=metadata,
248            file_info=file_info,
249            parsed_data=parsed_data,
250            time_offset=time_offset,
251        )
252
253    def _derive_data(self) -> Tuple[np.ndarray, np.ndarray, float]:
254        """
255        Calculate the SplineMetric on the data of the parent Modality.
256        """
257        message = ("Currently SplineMetric Modalities have to be "
258                   "calculated at instantiation time. In: " + self.name)
259        raise NotImplementedError(message)
260
261    def get_meta(self) -> dict:
262        # This conversion is done to keep nestedtext working.
263        meta = self.metadata.model_dump()
264        return meta
265
266    @property
267    def name(self) -> str:
268        """
269        Identity, metric, and parent data class.
270
271        The name will be of the form
272        'PD [metric name] on [data modality class name]'.
273
274        This overrides the default behaviour of Modality.name.
275        """
276        return SplineMetric.generate_name(self.metadata)
